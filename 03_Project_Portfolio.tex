\newlength{\projimgH}

% Single project card: {title}{url-or-empty}{description}{image-path}
\newcommand{\projcard}[4]{%
  \begin{tcolorbox}[projcard, equal height group=projrow\theprojrow]
    {\small\bfseries\color{highlightcolor}%
      \ifx\relax#2\relax#1\else\href{#2}{#1}\fi}\par
    \vspace{3pt}
    {\footnotesize #3}\par
    \vfill                      
    \noindent
    \begin{minipage}[c][\projimgH][c]{\linewidth}
      \centering
      \includegraphics[width=\linewidth, height=\projimgH,
        keepaspectratio=true]{#4}
    \end{minipage}
  \end{tcolorbox}%
}

% Two-column row of project cards
% Usage: \projRowTwo{t1}{u1}{d1}{i1}{t2}{u2}{d2}{i2}
\newcommand{\projRowTwo}[8]{%
  \noindent
  \begin{minipage}[t]{0.4955\linewidth}
     \vspace{0pt}
    \projcard{#1}{#2}{#3}{#4}
  \end{minipage}\hfill
  \begin{minipage}[t]{0.4955\linewidth}
     \vspace{0pt}
    \projcard{#5}{#6}{#7}{#8}
  \end{minipage}%
}
